To replace this demo data with real Stripe revenue, choose one of the options below.

Option A — Vercel Deployment (recommended)
  • Deploy to Vercel: npm i -g vercel && vercel
  • Set environment variable: STRIPE_SECRET_KEY=sk_live_xxx

Vercel Serverless Function — /api/revenue.js
// /api/revenue.js — Vercel Serverless Function
// Deploy to Vercel, set STRIPE_SECRET_KEY env var
import Stripe from 'stripe';
const stripe = new Stripe(process.env.STRIPE_SECRET_KEY);

export default async function handler(req, res) {
  const now = Math.floor(Date.now() / 1000);
  const thirtyDaysAgo = now - (30 * 24 * 60 * 60);

  const [charges, subscriptions] = await Promise.all([
    stripe.charges.list({ created: { gte: thirtyDaysAgo }, limit: 100 }),
    stripe.subscriptions.list({ status: 'active', limit: 100 })
  ]);

  const revenue = charges.data
    .filter(c => c.paid && !c.refunded)
    .reduce((sum, c) => sum + c.amount, 0) / 100;

  res.json({
    revenue: revenue.toFixed(2),
    orders: charges.data.length,
    mrr: subscriptions.data.reduce(
      (sum, s) => sum + s.items.data[0].price.unit_amount, 0
    ) / 100,
    charges: charges.data.slice(0, 10)
  });
}
